```html

  "apiConfig": {
    "API_HOST": "",
    "WARP_10_HOST": ""
    "API_OAUTH_TOKEN": "",
    "API_OAUTH_TOKEN_SECRET": "",
    "OAUTH_CONSUMER_KEY": "",
    "OAUTH_CONSUMER_SECRET": "",
  },
  "ownerId": ""

</cc-smart-container>
```
